            #<Relic Pool Setup>

                # Note that the Relic pools work totally differently than every other
                # item's pools.  There ARE some rarity-level pools, but relics don't
                # actually work that way, and basically nothing uses them.  So what we do
                # is play some games with weights and just add in legendaries, etc, to the
                # "Reward" relic pool.  This means that Relics get their own custom
                # weights in our code, which won't actually match everything else exactly.

                set GD_Itempools.ArtifactPools.Pool_ArtifactsReward BalancedItems
